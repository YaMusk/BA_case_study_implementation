Definition Context := form -> Prop.

Definition isClosed (Γ : Context) :=
  forall s, (Sigma (L : list form), prod (sublist L Γ) (L |- s)) -> Γ s.

Class theory := mkTheory {
                  Γ : Context;
                  Γ_closed : isClosed Γ;
}.

Lemma dec_f (s1 s2 : form) :
  {s1 = s2} + {s1 <> s2}.
Proof.
  decide equality.
  decide equality.
Qed.

Definition deriv' (P : Context) (s : form) := Sigma (L : list form), prod (sublist L P) (L |- s).
Definition deriv (T :theory) (s : form) := deriv' Γ s.
Notation "P |-p s" := (deriv' P s) (at level 70, right associativity).
Notation "T |-t s" := (deriv T s) (at level 70, right associativity).

Definition add (C : Context) (s : form) : Context := fun b => C b \/ b = s.

Definition isPrime (T : theory) := forall (s t : form), Γ (s ∨ t) -> Γ s \/ Γ t.


Class PCTheory := mkPrimeConsistentTheory {
                  Th : theory;
                  Th_prime : isPrime Th;
                  consistentT: ((Th |-t Bot) -> False);
}.

Notation "T |-pct s" := (deriv (@Th T) s) (at level 101, left associativity).

Definition tel (T : PCTheory) (s : form) := (@Γ (@Th T)) s.

Notation "s € T" := (tel T s) (at level 38, left associativity).

Definition TN (T : PCTheory) (s : form) := (N s) € T.

Definition R (T1 T2 : PCTheory) : Prop := forall s, s € T1 -> s € T2.
Definition E (T1 T2 : PCTheory) : Prop := (forall s, (TN T1 s) -> s € T2).


Lemma R_pre :
  PreOrder R.
Proof.
  split.
  - unfold Reflexive. intros T. unfold R. intuition.
  - unfold Transitive. intros T1 T2 T3. intros H1 H2. unfold R in *. firstorder.
Qed.

Lemma CM_concat (T1 T2 T3 : PCTheory) :
  R T1 T2 -> E T2 T3 -> E T1 T3.
Proof.
  intros H1 H2. unfold E in *. intros s H3. unfold R in H1.
  apply H2. unfold TN in *. intuition.
Qed.

Definition CM_val (T : PCTheory) (x : nat) := (var x) € T.

Lemma CM_persistent (T1 T2 : PCTheory) (x : nat) :
  R T1 T2 -> (CM_val T1 x) -> (CM_val T2 x).
Proof.
  intros H1 H2.
  unfold CM_val in *. unfold R in H1. intuition.
Qed.

Definition CanonicalModel : IMFrame.
apply mkIMFrame with (world := PCTheory) (cogn := R) (R := E) (val := CM_val).
- split.
  + unfold Reflexive. intros T. unfold R. intuition.
  + unfold Transitive. intros T1 T2 T3. intros H1 H2.
    unfold R in *. firstorder.
- apply CM_persistent.
- apply CM_concat.
Defined.

Lemma fault (L : list form) (s : form) :
  s el (remove dec_f s L) -> False.
Proof.
  intros H. induction L.
  - cbn in H. easy.
  - cbn in H. destruct (dec_f s a).
    + easy.
    + destruct H.
      * intuition.
      * easy.
Qed.

Lemma fault2 (L : list form) (s b : form) :
  s el (remove dec_f b L) -> s <> b -> s el L.
Proof.
  intros H H2. induction L.
  - cbn in H. easy.
  - cbn in H. destruct (dec_f b a).
    + right. apply IHL. easy.
    + destruct H.
      * left. easy.
      * right. firstorder.
Qed.

Lemma fault3 (L : list form) (s b : form) :
  s el L -> s <> b -> s el (remove dec_f b L).
Proof.
  intros H H2. induction L.
  - cbn in H. easy.
  - cbn. destruct (dec_f b a).
    + apply IHL. destruct H.
      * rewrite H in e. rewrite e in H2. easy.
      * easy.
    + destruct H.
      * left. easy.
      * right. apply IHL. easy.
Qed.

Lemma rev_ndII (A : list form) (s t : form) :
  A |- s ⊃ t -> s::A |- t.
Proof.
  intros H. apply (ndIE _ s).
  - apply (nd_weakening A); intuition.
  - apply ndA. intuition.
Qed.

Lemma CdA (T : theory) (s : form) :
  Γ s -> Γ |-p s.
Proof.
  intros H. exists [s]. split.
  - intros b H2. destruct H2 as [H2|H2].
    + rewrite <- H2. easy.
    + easy.
  - apply ndA. intuition.
Qed.

Lemma CdE (T : theory) (s : form) :
  Γ |-p Bot -> Γ |-p s.
Proof.
  intros [L H]. exists L. split.
  - intuition.
  - apply ndE. intuition.
Qed.

Lemma CdII (T : theory) (s t : form) :
  ((add Γ s) |-p t) -> (Γ |-p (s ⊃ t)).
Proof.
   intros H.
   destruct H as [lT lht]. unfold "|-p".
   exists (remove dec_f s lT). split.
   - destruct lht as [sub ded]. intros b H2. cbn in H2. specialize (sub b).
     destruct (dec_f s b).
     + subst b. apply fault in H2. easy.
     + apply (fault2 )in H2. 2: intuition. firstorder. rewrite H in n. easy.
   - destruct lht as [sub ded]. apply ndII. apply (nd_weakening lT). easy.
     intros b H. destruct (dec_f s b).
     + left. easy.
     + right. apply fault3. easy. firstorder.
Qed.

Lemma CdIE (T : theory) (s t : form) : (Γ |-p (s ⊃ t)) -> Γ |-p s -> Γ |-p t.
Proof.
  intros [L1 H1] [L2 H2].
  exists (L1++L2). split.
  - intros b H3. apply in_app_iff in H3. firstorder.
  - apply ndIE with s.
    + apply nd_weakening with L1. firstorder. intuition.
    + apply nd_weakening with L2. firstorder. intuition.
Qed.

Lemma CdOIl(T : theory) (s t : form) : Γ |-p s -> Γ |-p s ∨ t.
Proof.
  intros [L H]. exists L. split.
  - intuition.
  - apply ndOIl. intuition.
Qed.

Lemma CdOIr(T : theory) (s t : form) : Γ |-p t -> Γ |-p s ∨ t.
Proof.
  intros [L H]. exists L. split.
  - intuition.
  - apply ndOIr. intuition.
Qed.

Lemma CdAI (T : theory) (s t : form) : Γ |-p s -> Γ |-p t -> Γ |-p s ∧ t.
Proof.
  intros [L1 H1] [L2 H2]. exists (L1++L2). split.
  - intros b H3. apply in_app_iff in H3. firstorder.
  - apply ndAI.
    + apply nd_weakening with L1; intuition.
    + apply nd_weakening with L2; intuition.
Qed.

Lemma CdAEl (T : theory) (s t : form) : Γ |-p s ∧ t -> Γ |-p s.
Proof.
  intros [L H]. exists L. split.
  - intuition.
  - apply ndAEl with t. intuition.
Qed.

Lemma CdAEr (T : theory) (s t : form) : Γ |-p s ∧ t -> Γ |-p t.
Proof.
  intros [L H]. exists L. split.
  - intuition.
  - apply ndAEr with s. intuition.
Qed.

Definition LEM := forall P, P + (P -> False).

Lemma rev_contraposition (P Q : Prop) :
  LEM -> (not Q -> not P) -> P -> Q.
Proof.
  intros lem H p. destruct (lem Q).
  - easy.
  - exfalso. apply H; easy.
Qed.

Lemma asDeduction (w : PCTheory) (s : form) :
  s € w -> w |-pct s.
Proof.
  intros H. unfold "|-pct". exists [s]. split.
  - intros b. intros H2. destruct H2 as [H2|H2].
    + rewrite <- H2. easy.
    + contradiction.
  - apply ndA. firstorder.
Qed.

Fixpoint list_TN (A : list form) : list form :=
  match A with
  | nil => nil
  | a::A => (match a with N b => b::(list_TN A) | _ => (list_TN A) end) end.


Definition pred_TN (A : Context) : Context := fun s => A (N s).

Lemma member (A : Context) (s : form) :
  A (N s) -> (pred_TN A) s.
Proof.
  intros H. unfold pred_TN. easy.
Qed.

Lemma Artemov (A : Context) (s : form) :
(pred_TN A) |-p s -> A |-p N s.
Admitted.

Lemma lindenbaum (A : Context) (s : form) :
  ((A |-p Bot) -> False) -> ((A |-p s) -> False) -> exists (Δ : PCTheory),
      isPrime (@Th Δ) /\ (forall s, A s -> s € Δ) /\ (s € Δ -> False).
Admitted.

Lemma Tcomp (T : PCTheory) (s : form) :
  s € T -> T |-pct s.
Proof.
  intros H. unfold "|-pct". unfold tel in H. exists [s].
  split.
  - intros b. firstorder. rewrite <- H0. easy.
  - apply ndA. firstorder.
Qed.


Lemma truthLemma :
 LEM -> forall (s : form) (w : PCTheory), @evalK CanonicalModel s w <-> s € w.
Proof.
  intros lem s.
  induction s.
  - split.
    + intros H. cbn in H. easy.
    + intros H. unfold tel in H. assert(w |-pct Bot) by (apply (Tcomp w Bot H)).
      exfalso. apply consistentT. easy.
  - split.
    + intros H. unfold tel. cbn in H. easy.
    + intros H. unfold tel in H. cbn. firstorder.
  - split.
    + apply (rev_contraposition _ _ lem). intros H H2.
      cbn in H2.       destruct (lem (w |-pct s1 ⊃ s2)) as [?|condition].
      * destruct d as [L [H3 k]].
        contradict H. apply Γ_closed. exists L. easy.
      * assert(((add(@Γ (@Th w)) s1) |-p s2) -> False) as C2.
        { intros H3. apply condition. apply CdII. easy. }
        assert(((add(@Γ (@Th w)) s1) |-p Bot) -> False) as C1.
        { intros temp. destruct temp as [L [s d]]. apply C2.
          exists L. split. easy. apply ndE. easy. }
        destruct(lindenbaum (add(@Γ (@Th w)) s1) s2 C1 C2) as [Δ [L1 [L2 L3]]].
        assert(R w Δ) as RwΔ by firstorder.
        specialize (H2 Δ RwΔ).
        assert(s1 € Δ) as L.
        { firstorder. }
        apply IHs1 in L. apply H2 in L. firstorder.
    + intros H. cbn. intros w' Rww' H2.
      enough(@Γ (@Th w') |-p s2).
      { apply IHs2. apply Γ_closed in H0. firstorder. }
      apply (CdIE _ s1).
      * apply CdA. firstorder.
      * apply CdA. apply IHs1. easy.
  - split.
    + intros H. unfold tel.
      cbn in H. destruct H.
      * enough(s1 € w).
        { assert(@Γ (@Th w) |-p s1). {apply CdA. easy. }
          apply (CdOIl _ _ s2) in H1. apply Γ_closed in H1. easy. }
        apply IHs1. easy.
      * enough(s2 € w).
        { assert(@Γ (@Th w) |-p s2). {apply CdA. easy. }
          apply (CdOIr _ s1) in H1. apply Γ_closed in H1. easy. }
        apply IHs2. easy.
    + intros H. cbn. unfold tel in H.
      apply Th_prime in H. destruct H.
      * left. apply IHs1. firstorder.
      * right. apply IHs2. firstorder.
  - split.
    + intros H. unfold tel.
      enough((s1 € w /\ s2 € w)).
      { destruct H0 as [H1 H2]. apply Γ_closed. apply CdAI;
        apply CdA; easy. }
      cbn in H. destruct H as [H1 H2].
      split; firstorder.
    + intros H. cbn. apply asDeduction in H.
      split.
      * apply CdAEl in H. apply Γ_closed in H. firstorder.
      * apply CdAEr in H. apply Γ_closed in H. firstorder.
  - split.
    + apply (rev_contraposition _ _ lem). intros H H2. apply H.
      assert(((@Γ (@Th w)) |-p N s) -> False) as H3.
      { intros H3. apply H. apply Γ_closed in H3. firstorder. }
      
      set (LTN := pred_TN (@Γ (@Th w))).
      assert(LTN |-p s -> False).
      { intros S. apply Artemov in S. easy. }
      assert(LTN |-p Bot -> False) as Cond.
      { intros [L H4]. apply H0. exists L. split. intuition. apply ndE. intuition. }
      destruct(lindenbaum LTN s Cond H0) as [Δ [L1 [L2 L3]]].
      cbn in H2. assert(E w Δ).
      { unfold E. intros s' tn. unfold TN in tn. apply member in tn. firstorder. }
      specialize (H2 Δ H1). apply IHs in H2. firstorder.
    + intros H. cbn. intros w' Eww'.
      unfold E in Eww'. apply IHs. apply Eww'. firstorder.
Qed.

Definition l_to_C (A : list form) : Context := fun b => b el A.

Lemma completeness_of_IntK (A : list form) (s : form) :
  LEM -> (entails A s) -> A |- s.
Proof.
  intros lem H. destruct (lem (A |- s)).
  + easy.
  + exfalso.
    set (C := l_to_C A).
    assert(C |-p s -> False) as Cf.
    { intros [L H2]. apply f. apply nd_weakening with L. intuition. intuition. }
    assert(C |-p Bot -> False) as Cond.
    { intros [L H2]. apply Cf. exists L. split. intuition. apply ndE. intuition. }
    destruct(lindenbaum _ _ Cond Cf) as [Δ [H2 [H3 H4]]].
    enough(forall b, b el A -> @evalK CanonicalModel b Δ) as En.
    { specialize (H CanonicalModel Δ En). apply truthLemma in H; firstorder. }
    intros b belA. apply truthLemma. apply lem. firstorder.
Qed.
