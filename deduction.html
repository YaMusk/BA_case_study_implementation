Inductive form: Type :=
| Bot
| var (x: nat)
| imp (s t : form)
| or (s t : form)
| and (s t : form)
| N (s : form).
Definition elim_form (p : form -> Type) :
  p Bot ->
  (forall x, p (var x)) ->
  (forall s t, p s -> p t -> p (imp s t)) ->
  (forall s t, p s -> p t -> p (or s t)) ->
  (forall s t, p s -> p t -> p (and s t)) ->
  (forall s, p s -> p (N s)) ->
  forall f, p f := fun pBot fVar fImp fOr fAnd fN =>
    fix Rec f := match f with
    | Bot => pBot
    | var x => fVar x
    | imp s t => fImp s t (Rec s) (Rec t)
    | or s t => fOr s t (Rec s) (Rec t)
    | and s t => fAnd s t (Rec s) (Rec t)
    | N s => fN s (Rec s) end.

Inductive bmform: Type :=

| Bot_b
| var_b (x: nat)
| imp_b (s t : bmform)
| or_b (s t : bmform)
| and_b (s t : bmform)
| Ni (s : bmform)
| Nm (s : bmform).
Notation "s ⊃* t" := (imp_b s t) (at level 51, right associativity).
Notation "-* s" := (s ⊃* Bot_b) (at level 35, right associativity).

Notation "s ∨* t" := (or_b s t) (at level 40, left associativity).
Notation "s ∧* t" := (and_b s t) (at level 40, left associativity).

Reserved Notation "A |-b s" (at level 70).

Inductive KAx : list form -> form -> Type :=
| ndA (A : list form) s: s el A -> A |- s
| ndE A s: A |- Bot -> A |- s
| ndII A s t: s::A |- t -> A |- s ⊃ t
| ndIE A s t: A |- s ⊃ t -> A |- s -> A |- t
| ndAI A s t: A |- s -> A |- t -> A |- s ∧ t
| ndAEl A s t: A |- s ∧ t -> A |- s
| ndAEr A s t: A |- s ∧ t -> A |- t
| ndOIl A s t: A |- s -> A |- s ∨ t
| ndOIr A s t: A |- t -> A |- s ∨ t
| ndOE A s t v: A |- s ∨ t -> A |- s ⊃ v -> A |- t ⊃ v -> A |- v
| K A s t : A |- N(s ⊃ t) ⊃ N s ⊃ N t
| Nec s : nil |- s -> (forall A, A |- N s)
where "A |- s" := (KAx A s).


Inductive S4KAx : list bmform -> bmform -> Type :=
| cdA (A : list bmform) s: s el A -> A |-b s
| cdE A s: (-* s)::A |-b Bot_b -> A |-b s
| cdII A s t: s::A |-b t -> A |-b s ⊃* t
| cdIE A s t: A |-b s ⊃* t -> A |-b s -> A |-b t

| cdAI A s t: A |-b s -> A |-b t -> A |-b s ∧* t
| cdAEl A s t: A |-b s ∧* t -> A |-b s
| cdAEr A s t: A |-b s ∧* t -> A |-b t
| cdOIl A s t: A |-b s -> A |-b s ∨* t
| cdOIr A s t: A |-b t -> A |-b s ∨* t
| cdOE A s t v: A |-b s ∨* t -> A |-b s ⊃* v -> A |-b t ⊃* v -> A |-b v

| TI A s: A |-b Ni s ⊃* s
| T4I A s: A |-b Ni s ⊃* Ni (Ni s)

| KI A s t : A |-b Ni (s ⊃* t) ⊃* Ni s ⊃* Ni t

| KM A s t : A |-b Nm (s ⊃* t) ⊃* Nm s ⊃* Nm t
| NecM s : nil |-b s -> (forall A, A |-b Nm s)
| NecI s : nil |-b s -> (forall A, A |-b Ni s)
where "A |-b s" := (S4KAx A s).

Fixpoint t' (s: form) : bmform :=
  match s with
    | Bot => Bot_b
    | var x => var_b x
    | s ⊃ s' => (Ni (t' s)) ⊃* (Ni (t' s'))
    | s ∨ s' => (Ni (t' s)) ∨* (Ni (t' s'))
    | s ∧ s' => (Ni (t' s)) ∧* (Ni (t' s'))
    | N s => (Nm (Ni (t' s)))
end.

Definition t (s: form) : bmform :=
  Ni (t' s).


Fixpoint lt (A : list form) : list bmform :=
   match A with
   | nil => nil
   | a::A => (t a)::lt(A)
end.

Lemma t_map (A : list form) (s : form) :
  s el A -> t s el lt A.
Proof.
  induction A.
  - firstorder.
  - intros H. destruct H as [H|H].
    + rewrite <- H. cbn. firstorder.
    + right. firstorder.
Qed.

Lemma nd_weakening (A : list form) (s : form) :
  A |- s -> forall B, A <<= B -> B |- s.
Proof.
  induction 1; intros B TM.
  - apply ndA. intuition.
  - apply ndE. firstorder.
  - apply ndII. assert(s :: A <> []) by congruence. firstorder.
  - apply (ndIE _ s); firstorder.
  - apply ndAI; firstorder.
  - apply (ndAEl _ _ t0). firstorder.
  - apply (ndAEr _ s). firstorder.
  - apply ndOIl. firstorder.
  - apply ndOIr. firstorder.
  - apply (ndOE _ s t0); firstorder.
  - apply K.
  - apply Nec. firstorder.
Qed.

Lemma weakening (A : list bmform) (s : bmform) :
  A |-b s -> forall B, A <<= B -> B |-b s.
Proof.
  induction 1; intros B TM.
  - apply cdA. intuition.
  - apply cdE. assert(-* s :: A <> []) by congruence. firstorder.
  - apply cdII. assert(s :: A <> []) by congruence. firstorder.
  - apply (cdIE _ s); firstorder.
  - apply cdAI; firstorder.
  - apply (cdAEl _ _ t0). firstorder.
  - apply (cdAEr _ s). firstorder.
  - apply cdOIl. firstorder.
  - apply cdOIr. firstorder.
  - apply (cdOE _ s t0); firstorder.
  - apply TI.
  - apply T4I.
  - apply KI.
  - apply KM.
  - apply NecM. firstorder.
  - apply NecI. firstorder.
Qed.

Lemma strengthening (s : bmform) :
nil |-b s -> forall A, A |-b s.
Proof.
  intros H A. apply (weakening nil _ H A). easy.
Qed.

Lemma NI (s : bmform) :
  nil |-b s -> nil |-b Ni s.
Proof.
  intros H. apply NecI. easy.
Qed.

Lemma NM (s : bmform) :
  nil |-b s -> nil |-b Nm s.
Proof.
  intros H. apply NecM. easy.
Qed.


Lemma cdII_rev (A : list bmform) (s t : bmform) :
  A |-b s ⊃* t -> s::A |-b t.
Proof.
  intros H.
  apply (cdIE _ s).
  - apply (weakening A); firstorder.
  - apply cdA. intuition.
Qed.

Fixpoint shift (A : list bmform) (s : bmform) : bmform :=
   match A with
   | nil => s
   | a::A => shift A (a ⊃* s)
end.

Definition iffT (A B: Type) : Type := (A -> B) * (B -> A).
Notation "X <=> Y" := (iffT X Y) (at level 95, no associativity).

Lemma shifting (A : list bmform) (s : bmform) :
  A |-b s <=> nil |-b shift A s.
Proof.
  revert s.
  induction A; intros s.
  - cbn. easy.
  - split.
    + intros H. cbn. apply cdII in H. apply IHA in H. easy.
    + intros H. cbn in H. apply IHA in H. apply cdII_rev. easy.
Qed.

Lemma appKI (A : list bmform) (s t : bmform) :
  A |-b Ni (s ⊃* t) -> A |-b Ni s ⊃* Ni t.
Proof.
  intros H.
  apply (cdIE _ (Ni (s ⊃* t))).
  apply KI. easy.
Qed.

Lemma appT (A : list bmform) (s : bmform) :
  A |-b Ni s -> A |-b s.
Proof.
  intros H.
  apply (cdIE _ (Ni s)).
  apply TI.
  easy.
Qed.

Fixpoint tail (a : bmform) (A : list bmform) :=
  match A with
  nil => a
| x::X => tail x X end.

Fixpoint ins (A : list bmform) (s : bmform) :=
  match A with
   | nil => s
   | a::A => ins A (Ni a ⊃* Ni s)
end.


Lemma modal_shifting (A : list form) (s : bmform) :
  lt A |-b s -> lt A |-b Ni s.
Proof.
Admitted.

Lemma N_con (s : bmform) :
  nil |-b Nm(Ni s) ⊃* Nm s.
Proof.
  apply (cdIE _ (Nm((Ni s) ⊃* s))).
  - apply KM.
  - apply NecM. apply TI.
Qed.

Lemma Tt (A : list bmform)(s : form) :
  A |-b t s ⊃* Ni(t s).
Proof.
  unfold t. apply T4I.
Qed.

Lemma N_re (s : form) :
  nil |-b Nm (t s) ⊃* Nm(Ni (t s)).
Proof.
  apply (cdIE _ (Nm((t s) ⊃* (Ni (t s))))).
  - apply KM.
  - apply NecM. apply Tt.
Qed.

Lemma K_s (s1 s2 : form) :
  nil |-b Nm(Ni(t s1 ⊃* t s2)) ⊃* Nm(Ni(t s1) ⊃* Ni(t s2)).
Proof.
  apply (cdIE _ (Nm (Ni(t s1 ⊃* t s2) ⊃* Ni(t s1) ⊃* Ni(t s2)))).
  - apply KM.
  - apply NecM. apply KI.
Qed.

Lemma transpose (A : list form) (s : form) :
  A |- s -> lt A |-b t s.
Proof.
  induction 1 as
  [ |A s H IH|A s1 s2 H IH|A s1 s2 H1 IH1 H2 IH2 |A s1 s2 H1 IH1 H2 IH2 |A s1 s2 H IH|A s1 s2 H IH |A s1 s2 H IH |
   A s1 s2 H IH |A s1 s2 s3 H1 IH1 H2 IH2 H3 IH3 |A s1 s2 |s H IH ].

  - apply cdA. apply t_map. easy.

  - apply cdE. unfold t in IH. unfold t' in IH.
    assert(lt A |-b Bot_b) as Ti.
    { apply (cdIE _ (Ni Bot_b)). apply TI. easy. }
    apply (weakening (lt A) Bot_b Ti). intuition.

  - change ((t s1)::lt A |-b t s2) in IH.
    apply cdII in IH.
    change (lt A |-b Ni (t s1 ⊃* t s2)).
    apply modal_shifting. easy.

  - change(lt A |-b Ni (Ni (t' s1) ⊃* Ni (t' s2))) in IH1.
    apply (cdIE _ _ (Ni (t' s1) ⊃* Ni (t' s2))) in IH1.
    + unfold t. unfold t in IH2. apply (cdIE _ (Ni (t' s1))); easy.
    + apply TI.

  - assert(lt A |-b t s1 ∧* t s2) as IH.
    { apply (cdAI _ _ _ IH1 IH2). }
    change (lt A |-b Ni (t s1 ∧* t s2)).
    apply modal_shifting. easy.

  - change (lt A |-b Ni (t s1 ∧* t s2)) in IH.
    assert (lt A |-b t s1 ∧* t s2) as IH'.
    { apply (cdIE _ (Ni (t s1 ∧* t s2)) _ (TI (lt A) (t s1 ∧* t s2)) IH). }
    apply (cdAEl _ _ (t s2)). easy.

  - change (lt A |-b Ni (t s1 ∧* t s2)) in IH.
    assert (lt A |-b t s1 ∧* t s2) as IH'.
    { apply (cdIE _ (Ni (t s1 ∧* t s2)) _ (TI (lt A) (t s1 ∧* t s2)) IH). }
    apply (cdAEr _ (t s1)). easy.

  - change (lt A |-b Ni (t s1 ∨* t s2)).
    apply modal_shifting.
    apply cdOIl. easy.

  - change (lt A |-b Ni (t s1 ∨* t s2)).
    apply modal_shifting.
    apply cdOIr. easy.

  - apply (cdOE _ (t s1) (t s2)).
    + change (lt A |-b Ni (t s1 ∨* t s2)) in IH1.
      apply appT. easy.
    + change (lt A |-b Ni (t s1 ⊃* t s3)) in IH2.
      apply appT. easy.
    + change (lt A |-b Ni (t s2 ⊃* t s3)) in IH3.
      apply appT. easy.

    apply modal_shifting.
    change(lt A |-b t (N (s1 ⊃ s2)) ⊃* Ni ((t(N s1)) ⊃* Ni(Nm (t s2)))).
    apply cdII.

|-b Nm (t s2)).
    apply (cdIE _ ((Nm (t s1)))). 2: apply cdII_rev. 2: apply TI.

    apply (weakening (Ni (Nm (Ni (t s1 ⊃* t s2))) :: Nm (t s1) :: lt A)). 2: firstorder.
    apply (cdIE _ ((Nm (Ni (t s1 ⊃* t s2))))). 2: apply cdII_rev. 2: apply TI.

      apply (cdIE _ (Nm (Ni (t s1)) ⊃* Nm (Ni (t s2)))).

    + apply (cdIE _ (Nm (Ni (t s1 ⊃* t s2)))). apply cdA. firstorder. apply cdA. firstorder.
