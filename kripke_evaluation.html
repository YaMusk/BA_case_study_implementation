Definition iffT (A B: Type) : Type := (A -> B) * (B -> A).
Notation "X <=> Y" := (iffT X Y) (at level 95, no associativity).

Class IMFrame := mkIMFrame {
                            world : Type;
                            valuation := nat -> Prop;
                            cogn : world -> world -> Prop;
                            R : world -> world -> Prop;
                            val : world -> valuation;
                            
                            preorderCogn: PreOrder cogn;
                            persistence : forall (w v : world) (i : nat), cogn w v -> val w i -> val v i;
                            concat : forall (w1 w2 w3 : world), cogn w1 w2 -> R w2 w3 -> R w1 w3;
                            
                            
}.

Notation "w ≤ v" := (cogn w v) (at level 38, left associativity).

Fixpoint evalK {M: IMFrame} (s: form) : world -> Prop :=
    match s with
    | Bot => (fun x => False)
    | var y => (fun x => val x y)
    | s ⊃ t => (fun x => forall y, x ≤ y -> (evalK s y) -> (evalK t y))
    | s ∨ t => (fun y => evalK s y \/ evalK t y)
    | s ∧ t => (fun y => evalK s y /\ evalK t y)
    | N s => (fun x => forall y, R x y -> evalK s y)
  end.

Definition evalK' {M: IMFrame} (A: list form) :=
    fun w => forall s, (s el A) -> @evalK M s w.

Lemma monotonicityForK {M : IMFrame} (s : form) :
  forall (w v : world), w ≤ v -> evalK s w -> evalK s v.
Proof.
  intros w v H1 H2.
  induction s as [ | |s1 IH1 s2 IH2|s1 IH1 s2 IH2|s1 IH1 s2 IH2|s IH].
  - exfalso. firstorder.
  - cbn in H2. cbn. apply (persistence w v); easy.
  - cbn. cbn in H2. intros y H3 H4. apply H2.
    + apply preorderCogn with v; easy.
    + exact H4.
  - firstorder.
  - firstorder.
  - cbn in H2. cbn. intros u vRu. apply H2. apply (concat w v u); easy.
Restart.
  revert s.
  refine(elim_form _ _ _ _ _ _ _).
  - intros w v H1 H2.
    exfalso. firstorder.
  - intros x w v H1 H2.
    cbn in H2. cbn. apply (persistence w v); easy.
  - intros s t IH1 IH2. intros w v H1 H2.
    cbn. cbn in H2. intros y H3 H4. apply H2.
    + apply preorderCogn with v; easy.
    + exact H4.
  - intros s t IH1 IH2. intros w v H1 H2.
    firstorder.
  - firstorder.
  - intros s IH. intros w v H1 H2.
    cbn in H2. cbn. intros u vRu. apply H2. apply (concat w v u); easy.
Qed.

Definition entails A s :=
    forall (M : IMFrame),
      ((forall w,evalK' A w -> @evalK M s w)).

Notation "A ⊨ s" := (entails A s) (at level 30).

Lemma soundnessForK (A: list form) (s: form) :
  A |- s -> A ⊨ s.
Proof.
  intros H.
  induction H as [ | |A s t H1 IH|A s t H1 IH1 H2 IH2| | | | | |A s t u H1 IH1 H2 IH2 H3 IH3 |A s t| ].
  1-2, 5-9: firstorder.
  - intros M w Aw. intros v wkv H2. apply IH.
    intros b b_el_sA. destruct b_el_sA as [b_eq_s|b_el_A].
    + rewrite <- b_eq_s. easy.
    + specialize (Aw b b_el_A). apply (monotonicityForK b w v wkv). easy.
  - intros M w Aw. specialize (IH1 M w Aw). apply (IH1 w). apply preorderCogn. apply IH2. easy.


  - intros M w Aw. specialize (IH1 M w Aw). destruct IH1 as [wFs|wFt].
    + specialize (IH2 M w Aw). cbn in IH2. apply (IH2 w). apply preorderCogn. easy.
    + specialize (IH3 M w Aw). cbn in IH3. apply (IH3 w). apply preorderCogn. easy.

  - intros M w Aw. intros w' wkw' H. intros v w'kv H1. intros v' vRv'.
    specialize (H v').
    assert (R w' v') as w'Rv'.
    {
    apply (concat w' v v' w'kv vRv').
    }
    specialize(H w'Rv'). apply (H v'). apply preorderCogn. apply (H1 v'). easy.
  - intros M w Aw. cbn. intros v wRv. firstorder.
Qed.

Class BMFrame := mkBMFrame {
                            bmworld : Type;
                            bmvaluation := nat -> Prop;
                            Ri : bmworld -> bmworld -> Prop;
                            Rm : bmworld -> bmworld -> Prop;
                            bmval : bmworld -> bmvaluation;
                           
                            preorderRi: PreOrder Ri;
                            persistenceRi : forall (w v : bmworld) (i : nat), Ri w v -> bmval w i -> bmval v i;
}.

Fixpoint evalS4K {M: BMFrame} (s: bmform) : bmworld -> Prop :=
    match s with
    | Bot_b => (fun x => False)
    | var_b y => (fun x => bmval x y)
    | s ⊃* t => (fun x => (evalS4K s x -> evalS4K t x))

    | s ∨* t => (fun y => evalS4K s y \/ evalS4K t y)
    | s ∧* t => (fun y => evalS4K s y /\ evalS4K t y)

    | Ni s => (fun x => forall y, Ri x y -> evalS4K s y)
    | Nm s => (fun x => forall y, Rm x y -> evalS4K s y)
end.

Definition evalS4K' {M: BMFrame} (A: list bmform) :=
    fun w => forall s, (s el A) -> @evalS4K M s w.

Definition entailsS4k A s :=
    forall (M : BMFrame),
      ((forall w,evalS4K' A w -> @evalS4K M s w)).

Notation "A ⊨* s" := (entailsS4k A s) (at level 30).


Axiom xm : forall (M : BMFrame) w s,
  evalS4K s w \/ evalS4K (-* s) w.

Lemma soundnessForS4K (A: list bmform) (s: bmform) :
  A |-b s -> A ⊨* s.
Proof.
  induction 1 as
  [A s H|A s H IH|A s t H IH|A s t H1 IH1 H2 IH2|? |? |? |? |?
  |A s t r H1 IH1 H2 IH2 H3 IH3|A s|A s|A s t|A s t|s H IH|s H IH]; intros M w Aw.
  1,5-9,14:firstorder.
  - assert(evalS4K s w \/ evalS4K (-* s) w) as XM by apply (xm M w s).
    destruct XM as [XM|XM].
    + easy.
    + specialize (IH M w).
      assert(evalS4K' (-* s :: A) w) as H2.
      {
        intros b H3. destruct H3 as [H3|H3].
        * rewrite <- H3. easy.
        * specialize (Aw b). apply Aw. easy.
      }
      exfalso. firstorder.
  - cbn. intros Ws.
    specialize (IH M w). apply IH.
    intros b H1. cbn in H1. destruct H1 as [H1|H1].
    + rewrite <- H1. easy.
    + apply Aw. easy.
  - specialize (IH1 M w Aw). cbn in *. apply IH1.
    specialize (IH2 M w). firstorder.

  - specialize (IH1 M w Aw). cbn in IH1. destruct IH1 as [IH1|IH1].
    + specialize (IH2 M w Aw). firstorder.
    + firstorder.

  - cbn. intros H. apply H. apply preorderRi.
  - cbn. intros H. intros v wRv. intros u vRu. apply H. apply preorderRi with v; easy.
  - cbn. intros H1 H2. intros v wRv. apply H1. easy. apply H2. easy.
  - intros v _. firstorder.
  - intros v _. firstorder.
Qed.

Definition sigma (M: IMFrame) : BMFrame :=
  mkBMFrame world cogn R val preorderCogn persistence.

Lemma preserve_sigma (M : IMFrame) (w : world) (s : form) :
   @evalK M s w <-> @evalS4K (sigma M) (t s) w.
Proof.
  revert w.
  induction s as [ |x |s1 IH1 s2 IH2|s1 IH1 s2 IH2|s1 IH1 s2 IH2 |s IH]; intros w.
  - split; intros H.
    + exfalso. cbn in H. easy.
    + exfalso. cbn in H. apply (H w). apply preorderCogn.
  - split; intros H.
    + intros v wkv. apply (monotonicityForK (var x) w v wkv) in H. cbn in H. cbn. easy.
    + cbn in H. cbn. apply (H w). apply preorderCogn.
  - split; intros H.
    + intros v wkv. intros H2.
      apply (IH2 v). apply H. exact wkv. apply (IH1 v). easy.
    + intros v wkv. intros H2.
      apply (IH2 v). cbn in H. firstorder.
  - split; intros H.
    + cbn in H. destruct H as [H|H].
      * cbn. intros v wkv. left. apply IH1. apply (monotonicityForK _ w v wkv). easy.
      * cbn. intros v wkv. right. apply IH2. apply (monotonicityForK _ w v wkv). easy.
    + cbn in H. assert(w ≤ w) as wkw by (apply preorderCogn).
      destruct (H w wkw) as [Hw|Hw].
      * cbn. left. apply IH1. easy.
      * cbn. right. apply IH2. easy.
  - split; intros H.
    + destruct H as [H1 H2]. cbn. intros v wkv. split.
      * apply IH1. apply (monotonicityForK _ w v wkv). easy.
      * apply IH2. apply (monotonicityForK _ w v wkv). easy.
    + cbn in H. assert(w ≤ w) as wkw by (apply preorderCogn).
      destruct (H w wkw) as [Hw1 Hw2]. cbn. firstorder.

  - split; intros H.
    + cbn. intros w' wkw'. intros v w'Rv. apply IH. apply H. apply (concat w w' v); easy.
    + intros v wRv. apply IH. cbn in H.
      assert(w ≤ w) as wkw by (apply preorderCogn). specialize (H w wkw v wRv). easy.
Qed.


Definition star (BM : BMFrame) : bmworld -> bmworld -> Prop :=
  fun (x y : bmworld) => exists (a b : bmworld), Ri x a /\ Rm a b /\ Ri b y.

Lemma subset (BM : BMFrame) (w v : bmworld) :
  Rm w v -> star BM w v.
Proof.
  intros wRv. exists w,v. split. apply preorderRi. split. exact wRv. apply preorderRi.
Qed.

Definition starframe (BM : BMFrame) : BMFrame :=
  mkBMFrame bmworld Ri (star BM) bmval preorderRi persistenceRi.

Notation "BM '*'" := (starframe BM) (at level 38, left associativity).

Lemma equ1 (BM : BMFrame) (w v : bmworld) :
(star BM) w v <-> (exists a, Ri w a /\ (star BM) a v).
Proof.
  split.
  - intros H. exists w. split.
    + apply preorderRi.
    + easy.
  - intros [a [H [b [c [H1 [H2 H3]]]]]]. exists b,c.
    split. apply preorderRi with a; easy. split; easy.
Qed.

Lemma equ2 (BM : BMFrame) (w v : bmworld) :
(star BM) w v <-> (exists a, (star BM) w a /\ Ri a v).
Proof.
  split.
  - intros H. exists v. split.
    + easy.
    + apply preorderRi.
  - intros [a [wRa akv]]. destruct wRa as [b [c [wkb [bRc cka]]]].
    exists b,c. split. easy. split. easy. apply preorderRi with a; easy.
Qed.

Definition Mix1 (p : bmform) := ((Ni (Nm p)) ⊃* Nm p) ∧* ((Nm p) ⊃* Ni (Nm p)).
Definition Mix2 (p : bmform) := ((Nm (Ni p)) ⊃* Nm p) ∧* ((Nm p) ⊃* Nm (Ni p)).

Lemma Mix (BM : BMFrame) (p : bmform) :
  forall (w : bmworld), @evalS4K (BM *) (Mix1 p) w /\ @evalS4K (BM *) (Mix2 p) w.
Proof.
  intros w. split.
  - cbn. split.
    + intros H v wR'v. apply (H w). apply preorderRi. easy.
    + intros H v wkv u vR'u. apply (H u). apply (equ1 BM w u).
      exists v; easy.
  - cbn. split.
    + intros H v wR'v. apply (H v). easy. apply preorderRi.
    + intros H v wR'v u vku. apply (H u). apply (equ2 BM w u).
      exists v; easy.
Qed.

Lemma preserve_star (BM : BMFrame) (w : bmworld) (s : form) :
  @evalS4K BM (t s) w <-> @evalS4K (BM *) (t s) w.
Proof.
  revert w. induction s as [ | | |s1 IH1 s2 IH2| |s IH].
  1-3,5:firstorder.

  - intros w. split.
    + intros H. intros v wkv. cbn in H. destruct (H v wkv) as [Hv|Hv].
      * left. apply IH1. easy.
      * right. apply IH2. easy.
    + intros H. intros v wkv. cbn in H. destruct (H v wkv) as [Hv|Hv].
      * left. apply IH1. easy.
      * right. apply IH2. easy.
  - intros w. split.
    + intros H. intros v wkv. intros u vR'u. destruct vR'u as [a [b [vka [aRb bku]]]].
      apply IH. cbn. cbn in H. intros u' uku'. cbn in H.
      assert(Ri w a) as wka by (apply preorderRi with v; easy).
      assert(Ri b u') as bku' by (apply preorderRi with u; easy).
      apply (H a wka b aRb u' bku').
    + intros H. intros v wkv. intros u vRu. apply IH. intros u' uku'. cbn in H. apply (H v wkv u).
      * apply subset. easy.
      * easy.
Qed.

Lemma concat_star (BM : BMFrame) :
  forall (w1 w2 w3 : bmworld), Ri w1 w2 -> (star BM) w2 w3 ->(star BM) w1 w3.
Proof.
  intros w v u. intros wkv [a [b [vka [aRb bku]]]].
  exists a,b. split. apply preorderRi with v; easy. split; easy.
Qed.

Definition rho (BM : BMFrame) : IMFrame :=
  mkIMFrame bmworld Ri (star BM) bmval preorderRi persistenceRi (concat_star BM) .

Lemma preserve_rho (BM : BMFrame) (w : bmworld) (s : form) :
  @evalS4K BM (t s) w <-> @evalK (rho BM) s w.
Proof.
  revert w. induction s as [ |x|s1 IH1 s2 IH2|s1 IH1 s2 IH2|s1 IH1 s2 IH2|s IH]; intros w.
  - split; intros H; exfalso.
    + cbn in H. assert(Ri w w) as wkw by (apply preorderRi). apply (H w wkw).
    + cbn in H. easy.
  - split; intros H.
    + cbn in *. assert(Ri w w) as wkw by (apply preorderRi). apply (H w wkw).
    + cbn. intros v wkv.
      apply (@monotonicityForK (rho BM) _ w v wkv) in H. cbn in H. easy.
  - split; intros H.
    + cbn. intros v wkv H1. cbn in H.
      apply IH2. apply IH1 in H1. intros u vku. cbn in H1.
      apply (H v wkv H1 u vku).
    + intros v wkv H1. cbn in H. apply IH2. apply H. easy. apply IH1. easy.
  - split; intros H.
    + cbn. cbn in H. assert(Ri w w) as wkw by (apply preorderRi).
      destruct (H w wkw) as [Hw|Hw].
      * left. apply IH1. easy.
      * right. apply IH2. easy.
    + intros v wkv. destruct H as [H|H].
      * left. apply IH1. apply (@monotonicityForK (rho BM) _ w v wkv) in H. easy.
      * right. apply IH2. apply (@monotonicityForK (rho BM) _ w v wkv) in H. easy.
  - split; intros H.
    + cbn in H. assert(Ri w w) as wkw by (apply preorderRi). destruct (H w wkw) as [H1 H2].
      split.
      * apply IH1. easy.
      * apply IH2. easy.
    + destruct H as [H1 H2]. cbn. intros v wkv. split.
      * apply (@monotonicityForK (rho BM) _ w v wkv) in H1. apply IH1 in H1. easy.
      * apply (@monotonicityForK (rho BM) _ w v wkv) in H2. apply IH2 in H2. easy.
  - split; intros H.
    + intros v wRv. apply IH. intros u vku.
      cbn in H. destruct wRv as [a [b [wka [aRb bkv]]]].
      assert(Ri w w) as wkw by (apply preorderRi).
      specialize (H a wka b aRb u). apply H. apply preorderRi with v; easy.
    + intros v wkv u vRu. apply IH. apply H.
      exists v,u. split. easy. split. easy. apply preorderRi.
Qed.

Lemma preserve_combined (IM : IMFrame) (w : world) (s : form) :
  @evalK IM s w <-> @evalK (rho (sigma IM)) s w.
Proof.
  split; intros H.
  - apply preserve_sigma in H. apply preserve_rho in H. easy.
  - apply preserve_rho in H. apply preserve_sigma in H. easy.
Qed.

Lemma semantic_equivalence' (s : form) :
  (forall M w, @evalK M s w) <-> forall M w, @evalS4K M (t s) w.
Proof.
  split.
  - intros H. intros BM w. apply preserve_rho. apply H.
  - intros H. intros IM w. apply preserve_sigma. apply H.
Qed.

Definition decidable (A : Type) : Type := A + (A -> False).

Lemma semantic_equivalence (A : list form) (s : form) :
  (A ⊨ s) <-> ((lt A) ⊨* (t s)).
Proof.
  split; intros H.
  - intros BM w Aw. specialize (H (rho BM) w). apply preserve_rho. apply H. intros b belA.
    specialize(Aw (t b)). apply t_map in belA. apply preserve_rho. apply Aw. easy.
  - intros M w Aw. specialize (H (sigma M) w). apply preserve_sigma. apply H. intros b belA.
    unfold evalK' in Aw. apply in_map_iff in belA. destruct belA as [x [H2 H3]].
    rewrite <- H2. apply preserve_sigma. firstorder.
Qed.

Lemma preserve_decidability (A : list form) :
  (forall s, decidable(lt A ⊨* s)) -> (forall s, decidable(A ⊨ s)).
Proof.
  intros H s.
  - destruct (H (t s)) as [H'|H'].
    + left. intros IM w H2. apply preserve_sigma. apply H'. intros b H3.
      apply in_map_iff in H3 as [b' [H3 H4]]. rewrite <- H3. apply preserve_sigma. apply H2. easy.
    + right. intros H2. apply H'. intros BM w H3.
      apply preserve_rho. apply H2. intros b H4. apply preserve_rho. apply H3.
      apply t_map. easy.
Qed.
